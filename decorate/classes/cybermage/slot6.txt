actor CMHVBMG : CMWeapon
{ 
	Inventory.PickupMessage "You got the Heavy Blaster MG!"
	Weapon.SlotNumber 6
	Weapon.SelectionOrder 3000
	Weapon.AmmoType "Cell"
	Weapon.AmmoGive 40
	Weapon.AmmoUse 1
	Obituary "%o was incinerated by %k's Heavy Blaster MG."
	Tag "Heavy Blaster MG"	
	States 
	{ 
	Spawn: 
		CMHV F -1 
		Loop 
	Ready: 
		CMHV A 1 A_WeaponReady 
		Loop 
	Fire: 
		CMHV C 0 A_GiveInventory("SamsaraCyberMageHVBMGAttackHandler")
		CMHV C 2 BRIGHT A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		CMHV C 0 A_JumpIfNoAmmo("A")
		CMHV C 0 A_GiveInventory("SamsaraCyberMageHVBMGAttackHandler")
		CMHV C 2 BRIGHT A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		CMHV C 0 A_JumpIfNoAmmo("A")
		CMHV C 0 A_GiveInventory("SamsaraCyberMageHVBMGAttackHandler")
		CMHV D 2 BRIGHT A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		CMHV D 0 A_JumpIfNoAmmo("A")
		CMHV D 0 A_GiveInventory("SamsaraCyberMageHVBMGAttackHandler")
		CMHV D 2 BRIGHT A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		CMHV D 0 A_JumpIfNoAmmo("A")
		CMHV D 0 A_GiveInventory("SamsaraCyberMageHVBMGAttackHandler")
		CMHV D 2 BRIGHT A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
	A:	
		CMHV G 2 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		CMHV E 2 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		CMHV H 2 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		CMHV E 2 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		CMHV G 2 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		CMHV A 0 A_ReFire
		CMHV A 2 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		goto Ready 
	Hold:
		CMHV C 0 A_GiveInventory("SamsaraCyberMageHVBMGAttackHandler")
		CMHV C 2 BRIGHT A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		CMHV D 2 BRIGHT A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		CMHV G 2 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		CMHV A 0 A_Refire
		CMHV A 2 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		Goto Ready
	}
} 

actor SamsaraCyberMageHVBMGAttackHandler : Trigger
{
	States
	{
	// Ammo checking/taking stuff
	Pickup:
		TNT1 A 0 A_JumpIfInventory("SamsaraHasInfiniteAmmo",1,"PickupPlaySound")
		TNT1 A 0 A_JumpIfInventory("Cell",1,1)
		stop
		TNT1 A 0 A_TakeInventory("Cell",1)
		goto PickupPlaySound
	// Sound stuff
	PickupPlaySound:
		TNT1 A 0 A_PlaySound("CM/HVFire",CHAN_WEAPON)
	// Attack stuff
	Pickup2:
		TNT1 A 0 A_JumpIfInventory("SPModeOn",1,"PickupCoop")
		TNT1 A 0 A_JumpIfInventory("CoopModeOn",1,"PickupCoop")
	PickupDM:
		TNT1 A 0 A_FireCustomMissile("CMHVBall", 0, 1, 0, 0)
		goto PickupFinish
	PickupCoop:
		TNT1 A 0 A_FireCustomMissile("CMHVBall_2", 0, 1, 0, 0)
	PickupFinish:
		TNT1 A 0 //A_SetBlend("f38210",0.2,5)
		stop
	}
}
