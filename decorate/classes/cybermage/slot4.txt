actor CMSMG : CMWeapon
{
	Inventory.PickupMessage "You got the SMG!"
	Weapon.SlotNumber 4
	Weapon.SelectionOrder 3200  
	Weapon.AmmoType "Clip" 
	Weapon.AmmoGive 30
	Weapon.AmmoUse 1
	Weapon.YAdjust 9 
	Decal "BulletChip"
	Obituary "%o was peppered by %k's SMG."
	Tag "SMG"  
	States
	{
	Spawn:
		CMMG F -1
		stop
	Ready:
		CMMG A 1 A_WeaponReady
		loop
	Fire:
		CMMG A 0 A_Jump(128,"Fire2") 
		CMMG A 0 A_PlaySound("CM/SMGFire",CHAN_WEAPON)
		CMMG C 0 bright A_GiveInventory("SamsaraCyberMageSMGAttackHandler")
		CMMG C 3 bright A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		CMMG D 0 bright A_GiveInventory("SamsaraCyberMageSMGAttackHandler")
		CMMG D 3 bright A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		CMMG E 0 bright A_GiveInventory("SamsaraCyberMageSMGAttackHandler")
		CMMG E 3 bright A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		CMMG E 0 A_ReFire
		goto Ready
	Fire2:	
		CMMG A 0 A_PlaySound("CM/SMGFire",CHAN_WEAPON)	
		CMMG H 0 bright A_GiveInventory("SamsaraCyberMageSMGAttackHandler")
		CMMG H 3 bright A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		CMMG I 0 bright A_GiveInventory("SamsaraCyberMageSMGAttackHandler")
		CMMG I 3 bright A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		CMMG J 0 bright A_GiveInventory("SamsaraCyberMageSMGAttackHandler")
		CMMG J 3 bright A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		CMMG J 0 A_ReFire
		goto Ready 
	}
}

actor SamsaraCyberMageSMGAttackHandler : Trigger
{
	States
	{
	// Ammo checking/taking stuff
	Pickup:
		TNT1 A 0 A_JumpIfInventory("SamsaraHasInfiniteAmmo",1,"PickupPlaySound")
		TNT1 A 0 A_JumpIfInventory("Clip",1,1)
		stop
		TNT1 A 0 A_TakeInventory("Clip",1)
		goto PickupPlaySound
	// Sound stuff
	PickupPlaySound:
		TNT1 A 0 //A_PlaySound("CM/SMGFire",CHAN_WEAPON)
	// Attack stuff
	Pickup2:
		TNT1 A 0 A_JumpIfInventory("SPModeOn",1,"PickupCoop")
		TNT1 A 0 A_JumpIfInventory("CoopModeOn",1,"PickupCoop")
	PickupDM:
		TNT1 A 0 A_FireBullets(5,1.5, 1, 10, "CMBulletPuff", FBF_NORANDOM|FBF_USEAMMO)
		stop
	PickupCoop:
		TNT1 A 0 A_FireBullets(5,1.5, 1, 12, "CMBulletPuffCoop", FBF_NORANDOM|FBF_USEAMMO)
		stop
	}
}
