actor CMRocketGun : CMWeapon
{
	Inventory.PickupMessage "You got the Bombardier!"
	Weapon.SlotNumber 5
	Weapon.SelectionOrder 3100  
	Weapon.AmmoType "RocketAmmo"
	Weapon.AmmoGive 8
	Weapon.AmmoUse 1
	Obituary "%o was destroyed by %k's Bombardier."
	Tag "Bombardier"
	+WEAPON.EXPLOSIVE
	+WEAPON.NOAUTOFIRE
	States
	{
	Spawn:
		CMRL F -1
		loop
	Ready:
		CMRL A 1 A_WeaponReady
		loop
	Fire:
		CMRL C 4 bright A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		CMRL D 0 A_GiveInventory("SamsaraCyberMageBombardierAttackHandler")
		CMRL D 4 bright A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		CMRL E 4 bright A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		Goto Ready  
	}
}

actor SamsaraCyberMageBombardierAttackHandler : Trigger
{
	States
	{
	// Ammo checking/taking stuff
	Pickup:
		TNT1 A 0 A_JumpIfInventory("SamsaraHasInfiniteAmmo",1,"PickupPlaySound")
		TNT1 A 0 A_JumpIfInventory("RocketAmmo",1,1)
		stop
		TNT1 A 0 A_TakeInventory("RocketAmmo",1)
		goto PickupPlaySound
	// Sound stuff
	PickupPlaySound:
		TNT1 A 0 A_PlaySound("CM/RLFire",CHAN_WEAPON)
	// Attack stuff
	Pickup2:
		TNT1 A 0 A_JumpIfInventory("SPModeOn",1,"PickupCoop")
		TNT1 A 0 A_JumpIfInventory("CoopModeOn",1,"PickupCoop")
	PickupDM:
		TNT1 A 0 A_FireCustomMissile("CMRocket", 0, 1, 0, 0)
		goto PickupFinish
	PickupCoop:
		TNT1 A 0 A_FireCustomMissile("CMRocket_2", 0, 1, 0, 0)
	PickupFinish:
		TNT1 A 0 A_SetBlend("f38210",0.2,5)
		stop
	}
}
