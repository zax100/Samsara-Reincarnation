actor CMFusionGun : CMWeapon
{ 
	Inventory.PickupMessage "You got the Fusion Gun!"
	Weapon.SlotNumber 2
	Weapon.SelectionOrder 3400
	Weapon.AmmoType "Shell"
	Weapon.AmmoGive 10
	Weapon.AmmoUse 1
	Obituary "%o was eliminated by %k's Fusion Gun."
	Tag "Fusion Gun"
	+WEAPON.NOAUTOFIRE
	States 
	{ 
	Spawn: 
		CMFG J -1 
		Loop 
	Ready: 
		CMFG A 1 A_WeaponReady 
		Loop 
	Fire: 
		CMFG C 3 BRIGHT A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH) 
		CMFG B 0 A_GiveInventory("SamsaraCyberMageFusionGunAttackHandler")
		CMFG B 3 BRIGHT A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		CMFG A 5 A_WeaponReady(WRF_NOFIRE|WRF_NOSWITCH)
		goto Ready 
	}
}

actor SamsaraCyberMageFusionGunAttackHandler : Trigger
{
	States
	{
	// Ammo checking/taking stuff
	Pickup:
		TNT1 A 0 A_JumpIfInventory("SamsaraHasInfiniteAmmo",1,"PickupPlaySound")
		TNT1 A 0 A_JumpIfInventory("Shell",1,1)
		stop
		TNT1 A 0 A_TakeInventory("Shell",1)
		goto PickupPlaySound
	// Sound stuff
	PickupPlaySound:
		TNT1 A 0 A_PlaySound("CM/FGFire",CHAN_WEAPON)
	// Attack stuff
	Pickup2:
		TNT1 A 0 A_JumpIfInventory("SPModeOn",1,"PickupCoop")
		TNT1 A 0 A_JumpIfInventory("CoopModeOn",1,"PickupCoop")
	PickupDM:
		TNT1 A 0 A_FireCustomMissile("CMFusionBall", 0, 1, 0, 0)
		goto PickupFinish
	PickupCoop:
		TNT1 A 0 A_FireCustomMissile("CMFusionBall_2", 0, 1, 0, 0)
	PickupFinish:
		TNT1 A 0 A_SetBlend("20e718",0.2,5)
		stop
	}
}
