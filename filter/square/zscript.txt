// Since the player's radius is smaller in TAOS, this handler made by Ivory Duke
// is needed to make certain sections of the game playable in Samsara
// Thanks to Ivory Duke for creating the code

Class SamsaraTAOSHandler : EventHandler
{
    Override void PlayerEntered(PlayerEvent e)
    {
        if(e.PlayerNumber >= 0 && PlayerInGame[e.PlayerNumber] && Players[e.PlayerNumber].Mo)
        {
            Players[e.PlayerNumber].Mo.A_SetSize(10, 48, True);
        }
    }
  
    Override void PlayerRespawned(PlayerEvent e)
    {
        if(e.PlayerNumber >= 0 && PlayerInGame[e.PlayerNumber] && Players[e.PlayerNumber].Mo)
        {
            Players[e.PlayerNumber].Mo.A_SetSize(10, 48, True);
        }
    }
}
